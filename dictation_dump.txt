            return $matches[1] . '_' . $matches[2];
        }
        return $userAgent;
    }

    private function isSignificantFingerprintChange($stored, $current) {
        // If we don't have both fingerprints, no comparison possible
        if (!is_array($stored) || !is_array($current)) {
            return false;
        }

        $differences = 0;
        $totalChecks = 0;

        // Check browser and version (high weight)
        if (isset($stored['user_agent_base'], $current['user_agent_base'])) {
            $totalChecks += 3;
            $oldBrowser = explode('_', $stored['user_agent_base']);
            $newBrowser = explode('_', $current['user_agent_base']);
            
            if ($oldBrowser[0] !== $newBrowser[0]) {
                $differences += 3; // Different browser is a major change
            } elseif (isset($oldBrowser[1], $newBrowser[1]) && 
                abs((int)$newBrowser[1] - (int)$oldBrowser[1]) > 5) {
                $differences += 2; // Major version change
            }
        }

        // Check platform (high weight)
        if (isset($stored['platform'], $current['platform'])) {
            $totalChecks += 3;
            if ($stored['platform'] !== $current['platform']) {
                $differences += 3;
            }
        }

        // Check screen properties (medium weight)
        if (isset($stored['resolution'], $current['resolution'])) {
            $totalChecks += 2;
            if ($stored['resolution'] !== $current['resolution']) {
                $differences += 2;
            }
        }

        if (isset($stored['color_depth'], $current['color_depth'])) {
            $totalChecks += 2;
            if ($stored['color_depth'] !== $current['color_depth']) {
                $differences += 2;
            }
        }

        // Check CPU cores (medium weight)
        if (isset($stored['cores'], $current['cores'])) {
            $totalChecks += 2;
            if ($stored['cores'] !== $current['cores']) {
                $differences += 2;
            }
        }

        // Check headers order (low weight but good for distinguishing similar machines)
        if (isset($stored['headers_order'], $current['headers_order'])) {
            $totalChecks += 1;
            if ($stored['headers_order'] !== $current['headers_order']) {
                $differences += 1;
            }
        }

        // Check connection info (low weight)
        if (isset($stored['connection_info'], $current['connection_info'])) {
            $totalChecks += 1;
            if ($stored['connection_info'] !== $current['connection_info']) {
                $differences += 1;
            }
        }

        // Calculate the difference percentage
        if ($totalChecks === 0) return false;
        $differencePercentage = ($differences / $totalChecks) * 100;

        // Return true if more than 40% of weighted characteristics have changed
        return $differencePercentage > 40;
    }

    public function generateCsrfToken($sessionId) {
        return hash_hmac('sha256', $sessionId, $this->secretKey);
    }

    public function validateCsrfToken($token, $sessionId) {
        return hash_equals(
            $this->generateCsrfToken($sessionId),
            $token ?? ''
        );
    }

    public function validateSecureKey($providedKey, $expectedKey) {
        return $providedKey === $expectedKey;
    }

        $currentToken = $_COOKIE[$cookieName] ?? null;
        if (!$currentToken) {
            return null;
        }

        $currentTokenHash = hash('sha256', $currentToken);
        
        // Look through all tokens for this project
        foreach ($tokens[$projectId] as $token) {
            if ($token['hash'] === $currentTokenHash) {
                return $token;
            }
        }
        
        return null;
    }

    private function removeStoredToken($projectId) {
        $tokens = $this->getTokensFromFile();
        
        if (!isset($tokens[$projectId])) {
            return;
        }

        $cookieName = $this->getCookieName($projectId);
        $currentToken = $_COOKIE[$cookieName] ?? null;
        if (!$currentToken) {
            return;
        }

        $currentTokenHash = hash('sha256', $currentToken);
        
        // Remove only the specific token that matches
        $tokens[$projectId] = array_filter($tokens[$projectId], function($token) use ($currentTokenHash) {
            return $token['hash'] !== $currentTokenHash;
        });
        
        // Remove the project entry if no tokens left
        if (empty($tokens[$projectId])) {
            unset($tokens[$projectId]);
        }
        
        file_put_contents(self::TOKENS_FILE, json_encode($tokens, JSON_PRETTY_PRINT));
    }

    private function cleanupExpiredTokens(&$tokens) {
        $now = time();
        foreach ($tokens as $projectId => &$projectTokens) {
            // Remove expired tokens
            $projectTokens = array_filter($projectTokens, function($token) use ($now) {
                return $token['expiry'] > $now;
            });
            
            // Remove project entry if no valid tokens remain
            if (empty($projectTokens)) {
                unset($tokens[$projectId]);
            }
        }
    }

    private function getTokensFromFile() {
        $content = file_get_contents(self::TOKENS_FILE);
        return json_decode($content, true) ?? [];
    }

    private function generateRememberToken($projectId) {
        return hash_hmac('sha256', 
            $projectId . '|' . time() . '|' . uniqid('', true),
            $this->secretKey
        );
    }

    private function getCookieName($projectId) {
        return self::REMEMBER_COOKIE_PREFIX . $projectId;
    }

    private function getDeviceFingerprint() {
        // Get screen info via JavaScript and store in session
        $screenInfo = $_SESSION['screen_info'] ?? [];
        
        return [
            'user_agent_base' => $this->getBaseUserAgent($_SERVER['HTTP_USER_AGENT'] ?? ''),
            'accept_language' => $_SERVER['HTTP_ACCEPT_LANGUAGE'] ?? '',
            'platform' => $_SERVER['HTTP_SEC_CH_UA_PLATFORM'] ?? '',
            'color_depth' => $screenInfo['colorDepth'] ?? null,
            'resolution' => ($screenInfo['width'] ?? '0') . 'x' . ($screenInfo['height'] ?? '0'),
            'timezone' => $screenInfo['timezone'] ?? null,
            'cores' => $screenInfo['cores'] ?? null,
            'accept_headers' => [
                $_SERVER['HTTP_ACCEPT'] ?? '',
                $_SERVER['HTTP_ACCEPT_ENCODING'] ?? '',
                $_SERVER['HTTP_ACCEPT_LANGUAGE'] ?? ''
            ],
            'headers_order' => $this->getHeadersOrder(),
            'connection_info' => [
                'port' => $_SERVER['REMOTE_PORT'] ?? '',
                'protocol' => $_SERVER['SERVER_PROTOCOL'] ?? ''
            ]
        ];
    }

    private function getHeadersOrder() {
        $headers = getallheaders();
        return array_keys($headers);
    }

    private function getBaseUserAgent($userAgent) {
        // Extract just the browser name and major version
        if (preg_match('/(Firefox|Chrome|Safari|Edge|Opera)\/(\d+)/', $userAgent, $matches)) {
            return $matches[1] . '_' . $matches[2];
        }
        return $userAgent;
    }

    private function isSignificantFingerprintChange($stored, $current) {
        // If we don't have both fingerprints, no comparison possible
        if (!is_array($stored) || !is_array($current)) {
            return false;
        }

        $differences = 0;
        $totalChecks = 0;

        // Check browser and version (high weight)
        if (isset($stored['user_agent_base'], $current['user_agent_base'])) {
            $totalChecks += 3;
            $oldBrowser = explode('_', $stored['user_agent_base']);
            $newBrowser = explode('_', $current['user_agent_base']);
            
            if ($oldBrowser[0] !== $newBrowser[0]) {
                $differences += 3; // Different browser is a major change
            } elseif (isset($oldBrowser[1], $newBrowser[1]) && 
                abs((int)$newBrowser[1] - (int)$oldBrowser[1]) > 5) {
                $differences += 2; // Major version change
            }
        }

        // Check platform (high weight)
        if (isset($stored['platform'], $current['platform'])) {
            $totalChecks += 3;
            if ($stored['platform'] !== $current['platform']) {
                $differences += 3;
            }
        }

        // Check screen properties (medium weight)
        if (isset($stored['resolution'], $current['resolution'])) {
            $totalChecks += 2;
            if ($stored['resolution'] !== $current['resolution']) {
                $differences += 2;
            }
        }

        if (isset($stored['color_depth'], $current['color_depth'])) {
            $totalChecks += 2;
            if ($stored['color_depth'] !== $current['color_depth']) {
                $differences += 2;
            }
        }

        // Check CPU cores (medium weight)
        if (isset($stored['cores'], $current['cores'])) {
            $totalChecks += 2;
            if ($stored['cores'] !== $current['cores']) {
                $differences += 2;
            }
        }

        // Check headers order (low weight but good for distinguishing similar machines)
        if (isset($stored['headers_order'], $current['headers_order'])) {
            $totalChecks += 1;
            if ($stored['headers_order'] !== $current['headers_order']) {
                $differences += 1;
            }
        }

        // Check connection info (low weight)
        if (isset($stored['connection_info'], $current['connection_info'])) {
            $totalChecks += 1;
            if ($stored['connection_info'] !== $current['connection_info']) {
                $differences += 1;
            }
        }

        // Calculate the difference percentage
        if ($totalChecks === 0) return false;
        $differencePercentage = ($differences / $totalChecks) * 100;

        // Return true if more than 40% of weighted characteristics have changed
        return $differencePercentage > 40;
    }

    public function generateCsrfToken($sessionId) {
        return hash_hmac('sha256', $sessionId, $this->secretKey);
    }

    public function validateCsrfToken($token, $sessionId) {
        return hash_equals(
            $this->generateCsrfToken($sessionId),
            $token ?? ''
        );
    }

    public function validateSecureKey($providedKey, $expectedKey) {
        return $providedKey === $expectedKey;
    }
